    private void btnHapusActionPerformed(java.awt.event.ActionEvent evt) {                                         
        int pilihBaris = tabelData.getSelectedRow();
        if (pilihBaris != -1) {
            model.removeRow(pilihBaris);
            javax.swing.JOptionPane.showMessageDialog(this, "Data Terhapus!");
        } else {
            javax.swing.JOptionPane.showMessageDialog(this, "Pilih baris yang ingin dihapus!");
        }
    }                                        

    private void btnSimpanActionPerformed(java.awt.event.ActionEvent evt) {                                          
        String nama = jTextField1.getText().trim(); // .trim() menghapus spasi kosong di depan/belakang
    String jenis = jComboBox1.getSelectedItem().toString();
    
    // VALIDASI: Cek apakah nama barang kosong
    if (nama.isEmpty()) {
        javax.swing.JOptionPane.showMessageDialog(this, "Gagal! Nama barang tidak boleh kosong.");
        return; // Berhenti di sini, jangan lanjut simpan
    }

    // Tambah data ke tabel
    Object[] data = {nama, jenis};
    model.addRow(data);
    
    bersih();
    javax.swing.JOptionPane.showMessageDialog(this, "Data Berhasil Disimpan!");
    }                                         

    private void btnKeluarActionPerformed(java.awt.event.ActionEvent evt) {                                          
        System.exit(0);
    }                                         

    private void btnCariActionPerformed(java.awt.event.ActionEvent evt) {                                        
      String cari = jTextField2.getText().trim();
    boolean ditemukan = false;
    
    if (cari.isEmpty()) {
        javax.swing.JOptionPane.showMessageDialog(this, "Masukkan kata kunci pencarian!");
        return;
    }

    // Melakukan loop pada isi tabel
    for (int i = 0; i < tabelData.getRowCount(); i++) {
        // Kolom 0 adalah Nama Barang
        String namaDiTabel = tabelData.getValueAt(i, 0).toString();
        
        if (namaDiTabel.equalsIgnoreCase(cari)) {
            // Jika ketemu, pilih baris tersebut secara otomatis di tabel
            tabelData.setRowSelectionInterval(i, i);
            ditemukan = true;
            javax.swing.JOptionPane.showMessageDialog(this, "Data ditemukan pada baris ke-" + (i + 1));
            break; 
        }
    }

    if (!ditemukan) {
        javax.swing.JOptionPane.showMessageDialog(this, "Kata '" + cari + "' tidak ada di tabel.");
    }
    }                                       

    private void btnEditActionPerformed(java.awt.event.ActionEvent evt) {                                        
       int i = tabelData.getSelectedRow();
    
    if (i == -1) {
        javax.swing.JOptionPane.showMessageDialog(this, "Pilih baris di tabel yang ingin diedit!");
        return;
    }

    // Ambil data baru dari form inputan
    String namaBaru = jTextField1.getText();
    String jenisBaru = jComboBox1.getSelectedItem().toString();

    if (namaBaru.isEmpty()) {
        javax.swing.JOptionPane.showMessageDialog(this, "Isi Nama Barang baru di form atas!");
    } else {
        // Update data di baris yang sedang dipilih
        model.setValueAt(namaBaru, i, 0);
        model.setValueAt(jenisBaru, i, 1);
        javax.swing.JOptionPane.showMessageDialog(this, "Data Berhasil Diperbarui!");
        bersih();
    }
    }
